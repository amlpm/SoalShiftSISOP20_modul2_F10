```Javascript
void assignParam(int * num, int min, int max, char * str){
	if(isNumber(str)){
		*num = atoi(str);
		if(*num < min || *num > max){
			printError();
			exit(EXIT_FAILURE);
		}
	}else if(isStar(str)){
		*num = -1;
	}else{
		printError();
		exit(EXIT_FAILURE);
	}
}
```
